    <script>
        const toxinsData = [
            { name: "Tetrodotoxina (TTX)", filter: "membrana", target: "Canales de $Na^+$", effect: "Parálisis flácida. Impide la fase 0 del PA.", icon: "🐡" },
            { name: "Toxina Botulínica", filter: "placa", target: "Proteínas SNARE", effect: "Bloquea liberación de Acetilcolina. Parálisis.", icon: "🥫" },
            { name: "Digoxina", filter: "membrana", target: "Bomba $Na^+/K^+$", effect: "Aumenta contractilidad cardíaca.", icon: "❤️" },
            { name: "Curare", filter: "placa", target: "Receptor Nicotínico", effect: "Antagonista competitivo. Relajación muscular.", icon: "🌿" },
            { name: "Warfarina", filter: "hematologia", target: "Vitamina K", effect: "Inhibe factores II, VII, IX y X.", icon: "🩸" }
        ];

        const constantes = [
            { name: "Potasio ($K^+$)", val: "4.2", range: "3.8 - 5.0", letal: "L: < 1.5 | H: > 9.0" },
            { name: "pH Sanguíneo", val: "7.4", range: "7.35 - 7.45", letal: "L: < 6.9 | H: > 8.0" },
            { name: "Glucosa", val: "85", range: "70 - 110", letal: "L: < 20 | H: > 600" }
        ];

        let apChart;

        function init() {
            // Chart Potencial Acción
            const ctx = document.getElementById('actionPotentialChart').getContext('2d');
            apChart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: [0, 1, 2, 3, 4, 5, 6, 7, 8],
                    datasets: [{
                        label: 'mV',
                        data: [-90, -90, -65, 35, 10, -70, -95, -90, -90],
                        borderColor: '#0056FF',
                        borderWidth: 5,
                        fill: true,
                        backgroundColor: 'rgba(0, 86, 255, 0.05)',
                        tension: 0.4,
                        pointRadius: 0
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    scales: {
                        y: { min: -110, max: 50, grid: { color: '#f1f5f9' } },
                        x: { display: false }
                    },
                    plugins: { legend: { display: false } }
                }
            });

            renderToxins();
            renderConstantes();
            setupNav();
        }

        function updateChart(type) {
            let newData, text;
            if(type === 'normal') {
                newData = [-90, -90, -65, 35, 10, -70, -95, -90, -90];
                text = "<strong>Normal:</strong> Despolarización por $Na^+$ y repolarización por $K^+$.";
            } else if(type === 'ttx') {
                newData = [-90, -90, -90, -90, -90, -90, -90, -90, -90];
                text = "<strong>TTX:</strong> El canal de Sodio está bloqueado. No hay despolarización.";
            } else if(type === 'tea') {
                newData = [-90, -90, -65, 35, 35, 35, 35, 30, 20];
                text = "<strong>TEA:</strong> Bloqueo de Potasio. Repolarización extremadamente lenta.";
            }
            apChart.data.datasets[0].data = newData;
            apChart.update();
            document.getElementById('chart-explanation').innerHTML = text;
            renderMathInElement(document.getElementById('chart-explanation'));
        }

        function renderToxins(filter = 'all') {
            const grid = document.getElementById('toxins-grid');
            grid.innerHTML = toxinsData.filter(t => filter === 'all' || t.filter === filter).map(t => `
                <div class="bg-white p-6 rounded-3xl border border-gray-100 shadow-lg hover:shadow-xl transition-all">
                    <span class="text-4xl block mb-4">${t.icon}</span>
                    <h3 class="font-black text-lg text-brand-dark-blue mb-1">${t.name}</h3>
                    <p class="text-brand-blue text-xs font-bold uppercase mb-4">${t.target}</p>
                    <p class="text-gray-500 text-sm">${t.effect}</p>
                </div>
            `).join('');
            renderMathInElement(grid);
        }

        function renderConstantes() {
            document.getElementById('constantes-list').innerHTML = constantes.map(c => `
                <div>
                    <div class="flex justify-between mb-2 font-bold">
                        <span>${c.name}</span>
                        <span class="text-brand-blue">${c.val}</span>
                    </div>
                    <div class="w-full h-3 bg-gray-100 rounded-full overflow-hidden flex">
                        <div class="bg-red-400 h-full w-1/4"></div>
                        <div class="bg-emerald-400 h-full w-1/2"></div>
                        <div class="bg-red-400 h-full w-1/4"></div>
                    </div>
                    <div class="flex justify-between text-[10px] mt-1 font-bold text-gray-400">
                        <span>${c.letal.split('|')[0]}</span>
                        <span class="text-emerald-600">NORMAL: ${c.range}</span>
                        <span>${c.letal.split('|')[1]}</span>
                    </div>
                </div>
            `).join('');
            renderMathInElement(document.getElementById('constantes-list'));
        }

        function setupNav() {
            document.querySelectorAll('.nav-btn').forEach(btn => {
                btn.onclick = (e) => {
                    document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
                    btn.classList.add('active');
                    document.querySelectorAll('.content-section').forEach(s => s.classList.add('hide'));
                    document.getElementById(e.target.dataset.target).classList.remove('hide');
                }
            });
        }

        window.toggleAnswer = (id) => document.getElementById(id).classList.toggle('hidden');

        document.getElementById('btn-normal').onclick = () => updateChart('normal');
        document.getElementById('btn-ttx').onclick = () => updateChart('ttx');
        document.getElementById('btn-tea').onclick = () => updateChart('tea');

        window.onload = init;
    </script>
